//Working with Image and Image Collections
//Lets import Sentinel-2 image
//var image = ee.Image("COPERNICUS/S2/20220403T073611_20220403T080039_T36JTN");

var bandsIn = ee.List(['B2','B3','B4','B5','B6','B7','B8','B8A','B11','B12'])
//var bandsOut = ee.List(['blue','green','red','re1','re2','re3','nir1','nir2','swir1','swir2'])
// var wavelengths =([496.6,559,664.5,703.9,740.2,782.5,835.1,864.8,,1613.7,2202.4])
// var image = ee.Image("COPERNICUS/S2/20220403T073611_20220403T080039_T36JTN").select(bandsIn,bandsOut)
var S2a = ee.ImageCollection('COPERNICUS/S2_SR')
                  .filterBounds(extent_aoi)
                  .filterDate('2022-04-20', '2022-04-30')
                  .select(bandsIn)
                  .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE', 10));
print(S2a, 'Image Collection uMgeni 2022');
// print(S2a, size)
 var S2a_median = S2a.median()
                    .clip(extent_aoi); 
  //Print your Image to your console tab to inspect it
print(S2a_median, 'Median reduced Image uMgeni 2022');

//Add your Image as a map layer
var visParams = {'min': 400,'max': [4000,3000,3000],   'bands':'B8,B3,B2'};
Map.addLayer(S2a_median, visParams, 'S2 uMgeni 2022 Median');

//We have now prepared cloudfree Sentinel-2 Imagery for our research area and are therefore ready to start the classification.

//In the next step, we will have a look at how to sample training data needed for the LULC-Classification.
// Land cover classifcation
// Collect ground thruthing data/classes
var classMerge = Plantation.merge(water).merge(builtup).merge(grass);
print(classMerge, 'Merged Classes');

// // Now, we can finally start collecting training data based on our defined classes.
// // To do so, we will use .sampleRegions. Please check the documentation to find out more about how this works!
// // Also, keep in mind that .sampleRegions only works for Images, not for ImageCollections!

var training = S2a_median.sampleRegions({
  collection: classMerge,
  properties: ['landcover'],
  scale: 20
});
print(training, 'Training Data');
// // //Classification
// // //Classification - Classification and Regression Trees (CART)
// // //The only thing left now is to actually train the classifier with our training data.
// // //GEE offers various training methods, like for example CART (Classification and Regression Trees)- or Random Forest-Classifiers.
//Make sure to check out the documentation to find out more about their syntax!

//Let us start with a simple CART-Classifier./

//Example CART Classifier:
var cartclassifier = ee.Classifier.smileCart().train({
  features: training,
  classProperty: 'landcover'
});


//Run the classification
var cartclassified = S2a_median.classify(cartclassifier);

//Display classification
Map.addLayer(cartclassified,
{min: 0, max: 3, palette: ['red', 'blue', 'green','yellow']},
'CART classification');

// // //Classification - Random Forest
// // //In a very similar fashion, we can easily adjust our code to use a Random Forest classifier.

//Example RandomForest Classifier:
var rfclassifier = ee.Classifier.smileRandomForest(50).train({
  features: training,
  classProperty: 'landcover'
});
//Run the classification
var rfclassified = S2a_median.classify(rfclassifier);

//Display classification
Map.addLayer(rfclassified,
{min: 0, max: 3, palette: ['blue', 'green', 'orange','yellow']},
'RF classification');

// Train a minimum distance classifier (Mahalanobis distance metric) from the training sample.
var trainedClassifier = ee.Classifier.minimumDistance('mahalanobis').train({
  features: training,
  classProperty: 'landcover',
  inputProperties: S2a_median.bandNames()
});
// classify the image
var imgClassified = S2a_median.classify(trainedClassifier);
//Display classification
Map.addLayer(imgClassified,
{min: 0, max: 3, palette: ['blue', 'green', 'orange','yellow']},
'minimum distance classifier classification');
